package com.tnsif.sacnnerprogram;
// to demonstrate scanner program
public class Person {
	private String name;    // data member
	private int income;
	private int tax;
	
	// generate getter and setter method
	public String getName() {
		return name;
	}
	public void setName(String name) {
		this.name = name;
	}
	public int getIncome() {
		return income;
	}
	public void setIncome(int income) {
		this.income = income;
	}
	public int getTax() {
		return tax;
	}
	public void setTax(int tax) {
		this.tax = tax;
	}
	@Override
	public String toString() {
		return "Person [name=" + name + ", income=" + income + ", tax=" + tax + "]";
	}

	

}



package com.tnsif.sacnnerprogram;

public class Taxcal {

	public void caluculatetax(Person p) {
		
		if(p.getIncome()<=1600) {
			p.setTax(0);
		}
		else if(p.getIncome()>1600 && p.getIncome()<=15000) {
			p.setTax(5);
		}
		else {
			p.setTax(10);
		}
	}
}
